def cardioid(a=-np.pi, b=np.pi, n=200):
    phi = np.linspace(a, b, n, endpoint=True, dtype=np.float64)
    r = 1.0 + np.sin(phi)
    return phi, r

phi, r = cardioid(n=200)

def cardioidVelocity(a=1.0, left=-np.pi, right=np.pi, num=200):
    t = np.linspace(left, right, num)
    x = 2.0 * a * np.sin(t) * (2.0 * np.cos(t) - 1.0)
    y = 2.0 * a * (np.power(np.sin(t), 2) - np.power(np.cos(t), 2) + np.cos(t))
    
    for i in range(0, num):
        vec = (x[i], y[i])
        normalized_vec = vec / np.linalg.norm(vec)
        x[i] = vec[0]
        y[i] = vec[1]
        
    return x, y

t = np.linspace(-np.pi, np.pi, 200)
x, y = cardioidVelocity(1.0)
