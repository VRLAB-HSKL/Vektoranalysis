def helix(radius=1.0, height=2.0, left=0.0, right=1.0, num=200):
    t = np.linspace(left, right, num)
    x = radius * np.cos(2.0 * np.pi * t)
    y = radius * np.sin(2.0 * np.pi * t)
    z = height * t
    return x, y, z

x, y, z = helix(2.0)

def helixVelocity(radius=1.0, height=2.0, left=0.0, right=1.0, num=200):
    t = np.linspace(left, right, num)
    x = -2.0 * np.pi * radius * np.sin(2.0 * np.pi * t)
    y = 2.0 * np.pi * radius * np.cos(2.0 * np.pi * t)
    z = np.full(num, height)
    
    for i in range(0, num):
        vec = (x[i], y[i], z[i])
        normalized_vec = vec / np.linalg.norm(vec)
        x[i] = vec[0]
        y[i] = vec[1]
        z[i] = vec[2]
    
    return x, y, z

x, y, z = helixVelocity(2.0)

def helixAcceleration(radius=1.0, height=2.0, left=0.0, right=1.0, num=200):
    t = np.linspace(left, right, num)
    x = -4.0 * (np.pi * np.pi) * radius * np.cos(2.0 * np.pi * t)  #-2.0 * np.pi * radius * np.sin(2.0 * np.pi * t)
    y = -4.0 * (np.pi**2) * radius * np.sin(2.0 * np.pi * t) #2.0 * np.pi * radius * np.cos(2.0 * np.pi * t)
    z = np.full(num, 1.0)
    
    for i in range(0, num):
        vec = (x[i], y[i], z[i])
        normalized_vec = vec / np.linalg.norm(vec)
        x[i] = vec[0]
        y[i] = vec[1]
        z[i] = vec[2]
    
    return x, y, z

x, y, z = helixAcceleration(2.0)
